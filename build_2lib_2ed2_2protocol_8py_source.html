162  if isinstance(data, str):
163  data = data.encode()
164 
165  super().encode(self.start + data + self.end)
166  self.activity()

168  def decode(self, data):
169  if len(data) == 0:
170  return
171 
172  self._data += data
173 
174  if data[-1] == self.start[0]:
175  # Got partial escape code. Wait for more data.
176  return
177 
178  while True:
179  if not self._inMsg:
180  c = self._data.split(self.start, 1)
181  self.nonDebugData(c[0])
182 
183  if len(c) == 1:
184  # No start of message in here.
185  self._data = bytearray()
186  return
187  else:
188  # Keep rest of data, and continue decoding.
189  self._data = c[1]
190  self._inMsg = True
191  else:
192  c = self._data.split(self.end, 1)
193  if len(c) == 1:
194  # No end of message in here. Wait for more.
195  return
196  else:
197  # Got a full message.
198  # Remove \r as they can be inserted automatically by Windows.
199  # If \r is meant to be sent, escape it.
200  msg = c[0].replace(b'\r', b'')
201  self.logger.debug('extracted ' + str(bytes(msg)))
202  self.activity()
203  super().decode(msg)
204  self._data = c[1]
205  self._inMsg = False

209  m = super().mtu
210  if m == 0:
211  return 0
212  return max(1, m - len(start) - len(end))

261  super().__init__(**kwargs)

270  def decode(self, data):
271  if len(data) == 0:
272  return
273 
274  (seq, msg) = self.decode_seq(data)
275  if data[0] & self.reset_flag:
276  self._decode_seq = seq
277 
278  if self._req:
279  # We got a response, so the request should be finished now.
280  self._req = False
281  # This is the first part of the response.
282  self._decode_seq_start = self._decode_seq
283 
284  if seq == self._decode_seq:
285  self._decode_seq = self.nextSeq(self._decode_seq)
286  if len(msg) > 0:

288  super().decode(msg)
289  else:
290  self.logger.debug(f'unexpected seq {seq} instead of {self._decode_seq}; dropped')
291

293  self._syncing = False
294  for r in self._request:
295  super().encode(r)
296  self.activity()

298  def decode_seq(self, data):
299  seq = 0
300  if len(data) == 0:
301  raise ValueError
302  seq = data[0] & 0x3f
303  if data[0] & 0x40:
304  if len(data) == 1:
305  raise ValueError
306  seq = (seq << 7) | data[1] & 0x7f
307  if data[1] & 0x80:
308  if len(data) == 2:
309  raise ValueError
310  seq = (seq << 7) | data[2] & 0x7f
311  if data[2] & 0x80:
312  if len(data) == 3:
313  raise ValueError
314  seq = (seq << 7) | data[3] & 0x7f
315  if data[3] & 0x80:
316  raise ValueError
317  return (seq, data[4:])
318  else:
319  return (seq, data[3:])
320  else:
321  return (seq, data[2:])
322  else:
323  return (seq, data[1:])
324 
325  def encode_seq(self, seq):
326  if seq < 0x40:
327  return bytes([seq & 0x3f])
328  if seq < 0x2000:
329  return bytes([
330  0x40 | ((seq >> 7) & 0x3f),
331  seq & 0x7f])
332  if seq < 0x100000:
333  return bytes([
334  0x40 | ((seq >> 14) & 0x3f),
335  0x80 | ((seq >> 7) & 0x7f),
336  seq & 0x7f])
337  if seq < 0x8000000:
338  return bytes([
339  0x40 | ((seq >> 21) & 0x3f),
340  0x80 | ((seq >> 14) & 0x7f),
341  0x80 | ((seq >> 7) & 0x7f),
342  seq & 0x7f])
343  return self.encode_seq(seq % 0x8000000)

345  def encode(self, data):
346  if self._reset:
347  self._reset = False
348  self._sync()
349 
350  if not self._req:
351  self._request = []
352 # if self._encode_seq >= 0x2000:
353 # # Try to keep the seq value low.
354 # self._sync()
355  self._encode_seq_start = self._encode_seq
356 
357  self._req = True
358  self._encode_seq = self.nextSeq(self._encode_seq)
359 
360  # We cannot handle a wrap around of the seq, as a retransmit will be bogus.
361  # Make sure the MTU is big enough to transmit the full request in 255 chunks.
362  assert self._encode_seq != self._encode_seq_start
363 
364  request = self.encode_seq(self._encode_seq) + data
365  self._request.append(request)
366  if not self._syncing:
367  super().encode(request)
368  self.activity()

371  if self._syncing:
372  return

374  self._encode_seq = 0
375  super().encode(bytes([self.reset_flag]))

377  def timeout(self):
378  super().timeout()
379  if not self._req:
380  self._decode_seq = self._decode_seq_start
381  self.retransmit()

388  self.logger.debug('retransmit')
