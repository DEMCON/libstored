72 void AsciiEscapeLayer::decode(void* buffer, size_t len) {
73  char* p = static_cast<char*>(buffer);
74 
75  // Common case: there is no escape character.
76  size_t i = 0;
77  for(; i + 1 < len; i++)
78  if(unlikely(p[i] == Esc))
79  goto first_escape;
80 
81  // No escape characters.
82  base::decode(buffer, len);
83  return;
84 
85 first_escape:
86  // Process escape sequences in-place.
87  size_t decodeOffset = i;
88 
89 escape:
90  if(p[++i] == Esc)
91  p[decodeOffset] = (uint8_t)Esc;
92  else
93  p[decodeOffset] = (uint8_t)p[i] & (uint8_t)EscMask;
94  decodeOffset++;
95  i++;
96 
97  // The + 1 prevents it from trying to decode an escape character at the end.
98  for(; i + 1 < len; i++, decodeOffset++) {
99  if(unlikely(p[i] == Esc))
100  goto escape;
101  else
102  p[decodeOffset] = p[i];
103  }
104 
105  // Always add the last character (if any).
106  if(i < len)
107  p[decodeOffset++] = p[i];
108  base::decode(p, decodeOffset);
109 }

111 char AsciiEscapeLayer::needEscape(char c) {
112  if(!((uint8_t)c & (uint8_t)~(uint8_t)AsciiEscapeLayer::EscMask)) {
113  if(!m_all) {
114  // Only escape what conflicts with other protocols.
115  switch(c) {
116  case '\0':
117  case '\x11': // XON
118  case '\x13': // XOFF
119  case '\x1b': // ESC
120  case '\r':
121  // Do escape \r, as Windows may inject it automatically. So, if \r
122  // is meant to be sent, escape it, such that the client may remove
123  // all (unescaped) \r's automatically.
124  break;
125 
126  default:
127  // Don't escape.
128  return 0;
129  }
130  }
131  return (char)((uint8_t)c | 0x40u);
132  } else if(c == AsciiEscapeLayer::Esc) {
133  return c;
134  } else {
135  return 0;
136  }
137 }

139 void AsciiEscapeLayer::encode(void const* buffer, size_t len, bool last) {
140  uint8_t const* p = static_cast<uint8_t const*>(buffer);
141  uint8_t const* chunk = p;
142 
143  for(size_t i = 0; i < len; i++) {
144  char escaped = needEscape((char)p[i]);
145  if(unlikely(escaped)) {
146  // This is a to-be-escaped byte.
147  if(chunk < p + i)
148  base::encode(chunk, (size_t)(p + i - chunk), false);
149 
150  uint8_t const esc[2] = { AsciiEscapeLayer::Esc, (uint8_t)escaped };
151  base::encode(esc, sizeof(esc), last && i + 1 == len);
152  chunk = p + i + 1;
153  }
154  }
155 
156  if(likely(chunk < p + len) || (len == 0 && last))
157  base::encode(chunk, (size_t)(p + len - chunk), last);
158 }

195 void TerminalLayer::decode(void* buffer, size_t len) {
196  size_t nonDebugOffset = m_decodeState < StateDebug ? 0 : len;
197 
198  for(size_t i = 0; i < len; i++) {
199  char c = (static_cast<char*>(buffer))[i];
200 
201  switch(m_decodeState) {
202  case StateNormal:
203  if(unlikely(c == Esc))
204  m_decodeState = StateNormalEsc;
205  break;
206  case StateNormalEsc:
207  if(likely(c == EscStart)) {
208  nonDebugDecode(static_cast<char*>(buffer) + nonDebugOffset, i - nonDebugOffset - 1); // Also skip the ESC
209  m_decodeState = StateDebug;
210  nonDebugOffset = len;
211  } else
212  m_decodeState = StateNormal;
213  break;
214  case StateDebug:
215  if(unlikely(c == Esc))
216  m_decodeState = StateDebugEsc;
217  else
218  m_buffer.push_back(c);
219  break;
220  case StateDebugEsc:
221  if(likely(c == EscEnd)) {
222  base::decode(&m_buffer[0], m_buffer.size());
223  m_decodeState = StateNormal;
224  m_buffer.clear();
225  nonDebugOffset = i + 1;
226  } else {
227  m_decodeState = StateDebug;
228  m_buffer.push_back((char)Esc);
229  m_buffer.push_back(c);
230  }
231  break;
232  }
233  }
234 
235  if(nonDebugOffset < len)
236  nonDebugDecode(static_cast<char*>(buffer) + nonDebugOffset, len - nonDebugOffset);
237 }

353 void SegmentationLayer::decode(void* buffer, size_t len) {
354  if(len == 0)
355  return;
356 
357  char* buffer_ = static_cast<char*>(buffer);
358  if(!m_decode.empty() || buffer_[len - 1] != EndMarker) {
359  // Save for later packet reassembling.
360  if(len > 1) {
361  size_t start = m_decode.size();
362  m_decode.resize(start + len - 1);
363  memcpy(&m_decode[start], buffer_, len - 1);
364  }
365 
366  if(buffer_[len - 1] == EndMarker) {
367  // Got it.
368  base::decode(&m_decode[0], m_decode.size());
369  m_decode.clear();
370  }
371  } else {
372  // Full packet is in buffer. Forward immediately.
373  base::decode(buffer, len - 1);
374  }
375 }

377 void SegmentationLayer::encode(void const* buffer, size_t len, bool last) {
378  char const* buffer_ = static_cast<char const*>(buffer);
379 
380  size_t mtu = lowerMtu();
381  if(mtu == 0)
382  mtu = std::numeric_limits<size_t>::max();
383  else if(mtu == 1)
384  mtu = 2;
385 
386  while(len) {
387  size_t remaining = mtu - m_encoded - 1;
388  size_t chunk = std::min(len, remaining);
389 
390  if(chunk) {
391  base::encode(buffer_, chunk, false);
392  len -= chunk;
393  buffer_ += chunk;
394  }
395 
396  if(chunk == remaining && len) {
397  // Full MTU.
398  char cont = ContinueMarker;
399  base::encode(&cont, 1, true);
400  m_encoded = 0;
401  } else {
402  // Partial MTU. Record that we already filled some of the packet.
403  m_encoded += chunk;
404  }
405  }
406 
407  if(last) {
408  // The marker always ends the packet.
409  char end = EndMarker;
410  base::encode(&end, 1, true);
411  m_encoded = 0;
412  }
413 }

439 void ArqLayer::decode(void* buffer, size_t len) {
440  if(len == 0)
441  return;
442 
443  uint8_t* buffer_ = static_cast<uint8_t*>(buffer);
444  uint8_t flags = *buffer_;
445  uint32_t seq = decodeSeq(buffer_, len);
446  if(unlikely(flags & ResetFlag)) {
447  // Reset communication state.
448  m_decodeState = DecodeStateIdle;
449  m_encodeState = EncodeStateIdle;
450  m_encodeBuffer.clear();
451  m_encodeBufferSize = 0;
452  m_decodeSeq = nextSeq(seq);
453  m_encodeSeq = 1;
454  setPurgeableResponse(false);
455  uint8_t ack = ResetFlag;
456  base::encode(&ack, sizeof(ack), true);
457  m_encodeSeqReset = false;
458  }
459 
460  switch(m_decodeState) {
461  case DecodeStateDecoded:
462  stored_assert(m_encodeState == EncodeStateIdle || m_encodeState == EncodeStateUnbufferedIdle);
463 
464  if(seq == m_decodeSeq) {
465  // This is the next command. The previous one was received, apparently.
466  m_decodeState = DecodeStateIdle;
467  m_encodeState = EncodeStateIdle;
468  m_encodeBuffer.clear();
469  m_encodeBufferSize = 0;
470  setPurgeableResponse(false);
471  } else
472  // fall-through
473  case DecodeStateRetransmit:
474  if(seq == m_decodeSeqStart) {
475  // This seems to be a retransmit of the current command.
476  switch(m_encodeState) {
477  case EncodeStateUnbufferedIdle:
478  // We must reexecute the command. Reset the sequence number, as the content may change.
479  setPurgeableResponse(false);
480  m_decodeSeq = m_decodeSeqStart;
481  m_decodeState = DecodeStateIdle;
482  m_encodeSeqReset = true;
483  break;
484  case EncodeStateIdle:
485  // Wait for full retransmit of the command, but do not actually decode.
486  m_decodeState = DecodeStateRetransmit;
487  break;
488  default:
489  stored_assert(false); // NOLINT(hicpp-static-assert,misc-static-assert)
490  }
491  } // else: unexpected seq; ignore.
492  break;
493  default:;
494  }
495 
496  switch(m_decodeState) {
497  case DecodeStateRetransmit:
498  if(nextSeq(seq) == m_decodeSeq) {
499  // Got the last part of the command. Retransmit the response buffer.
500  for(std::vector<std::string>::const_iterator it = m_encodeBuffer.begin(); it != m_encodeBuffer.end(); ++it)
501  base::encode(it->data(), it->size(), true);
502  m_decodeState = DecodeStateDecoded;
503  }
504  break;
505  case DecodeStateIdle:
506  m_decodeSeqStart = m_decodeSeq;
507  m_decodeState = DecodeStateDecoding;
508  // fall-through
509  case DecodeStateDecoding:
510  if(likely(seq == m_decodeSeq)) {
511  // Properly in sequence.
512  base::decode(buffer_, len);
513  m_decodeSeq = nextSeq(m_decodeSeq);
514 
515  // Should not wrap-around during a request.
516  stored_assert(m_decodeSeq != m_decodeSeqStart);
517  }
518  break;
519  default:;
520  }
521 }

523 void ArqLayer::encode(void const* buffer, size_t len, bool last) {
524  if(m_decodeState == DecodeStateDecoding) {
525  // This seems to be the first part of the response.
526  // So, the request message must have been complete.
527  m_decodeState = DecodeStateDecoded;
528  stored_assert(m_encodeState == EncodeStateIdle || m_encodeState == EncodeStateUnbufferedIdle);
529  }
530 
531  switch(m_encodeState) {
532  case EncodeStateIdle:
533  case EncodeStateEncoding:
534  if(unlikely(m_maxEncodeBuffer > 0 && m_encodeBufferSize + len > m_maxEncodeBuffer)) {
535  // Overflow.
536  setPurgeableResponse();
537  }
538  break;
539  default:;
540  }
541 
542  uint8_t seq[4];
543  size_t seqlen = 0;
544 
545  switch(m_encodeState) {
546  case EncodeStateUnbufferedIdle:
547  case EncodeStateIdle:
548  seqlen = encodeSeq(m_encodeSeq, seq);
549  stored_assert(seqlen <= sizeof(seq));
550  m_encodeSeq = nextSeq(m_encodeSeq);
551  // Assume a wrap-around will be noticed, as it contains at least 128 MB of data.
552  if(m_encodeSeqReset) {
553  seq[0] = (uint8_t)(seq[0] | ResetFlag);
554  m_encodeSeqReset = false;
555  }
556  break;
557  default:;
558  }
559 
560  switch(m_encodeState) {
561  case EncodeStateUnbufferedIdle:
562  base::encode(seq, seqlen, false);
563  m_encodeState = EncodeStateUnbufferedEncoding;
564  // fall-through
565  case EncodeStateUnbufferedEncoding:
566  base::encode(buffer, len, last);
567  if(last)
568  m_encodeState = EncodeStateUnbufferedIdle;
569  break;
570 
571  case EncodeStateIdle:
572 #if STORED_cplusplus >= 201103L
573  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
574  m_encodeBuffer.emplace_back(reinterpret_cast<char*>(seq), seqlen);
575 #else
576  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
577  m_encodeBuffer.push_back(std::string(reinterpret_cast<char*>(seq), seqlen));
578 #endif
579  m_encodeState = EncodeStateEncoding;
580  m_encodeBufferSize += seqlen;
581  // fall-through
582  case EncodeStateEncoding:
583  stored_assert(!m_encodeBuffer.empty());
584  m_encodeBufferSize += len;
585  m_encodeBuffer.back().append(static_cast<char const*>(buffer), len);
586 
587  if(last) {
588  base::encode(m_encodeBuffer.back().data(), m_encodeBuffer.back().size(), true);
589  m_encodeState = EncodeStateIdle;
590  }
591  }
592 }

594 void ArqLayer::setPurgeableResponse(bool purgeable) {
595  bool wasPurgeable = m_encodeState == EncodeStateUnbufferedIdle || m_encodeState == EncodeStateUnbufferedEncoding;
596 
597  if(wasPurgeable == purgeable)
598  return;
599 
600  if(purgeable) {
601  // Switch to purgeable.
602  base::setPurgeableResponse(true);
603 
604  switch(m_encodeState) {
605  case EncodeStateEncoding: {
606  std::string const& s = m_encodeBuffer.back();
607  base::encode(s.data(), s.size(), false);
608  m_encodeState = EncodeStateUnbufferedEncoding;
609  break; }
610  case EncodeStateIdle:
611  m_encodeState = EncodeStateUnbufferedIdle;
612  break;
613  default:
614  stored_assert(false); // NOLINT(hicpp-static-assert,misc-static-assert)
615  }
616 
617  m_encodeBuffer.clear();
618  m_encodeBufferSize = 0;
619  } else {
620  // Switch to precious.
621  switch(m_encodeState) {
622  case EncodeStateUnbufferedEncoding:
623  // First part already sent. Can't switch to precious right now.
624  // Wait till start of next command.
625  break;
626  case EncodeStateUnbufferedIdle:
627  base::setPurgeableResponse(false);
628  m_encodeState = EncodeStateIdle;
629  break;
630  default:
631  stored_assert(false); // NOLINT(hicpp-static-assert,misc-static-assert)
632  }
633  }
634 }

650 uint32_t ArqLayer::decodeSeq(uint8_t*& buffer, size_t& len) {
651  uint32_t seq = 0;
652  uint8_t flag = 0x40;
653 
654  while(true) {
655  if(!len--)
656  return ~0u;
657  seq = (seq << 7u) | (*buffer & (flag - 1u));
658  if(!(*buffer++ & flag))
659  return seq;
660  flag = 0x80;
661  }
662 }

664 size_t ArqLayer::encodeSeq(uint32_t seq, void* buffer) {
665  uint8_t* buffer_ = static_cast<uint8_t*>(buffer);
666  if(seq < 0x40u) {
667  buffer_[0] = (uint8_t)(seq & 0x3fu);
668  return 1;
669  } else if(seq < 0x2000) {
670  buffer_[0] = (uint8_t)(0x40u | ((seq >> 7u) & 0x3fu));
671  buffer_[1] = (uint8_t)(seq & 0x7fu);
672  return 2;
673  } else if(seq < 0x100000) {
674  buffer_[0] = (uint8_t)(0x40u | ((seq >> 14u) & 0x3fu));
675  buffer_[1] = (uint8_t)(0x80u | ((seq >> 7u) & 0x7fu));
676  buffer_[2] = (uint8_t)(seq & 0x7fu);
677  return 3;
678  } else if(seq < 0x8000000) {
679  buffer_[0] = (uint8_t)(0x40u | ((seq >> 21u) & 0x3fu));
680  buffer_[1] = (uint8_t)(0x80u | ((seq >> 14u) & 0x7fu));
681  buffer_[2] = (uint8_t)(0x80u | ((seq >> 7u) & 0x7fu));
682  buffer_[3] = (uint8_t)(seq & 0x7fu);
683  return 4;
684  } else {
685  return encodeSeq(seq % 0x8000000, buffer);
686  }
687 }

861 void BufferLayer::encode(void const* buffer, size_t len, bool last) {
862  char const* buffer_ = static_cast<char const*>(buffer);
863 
864  size_t remaining = m_size - m_buffer.size();
865 
866  while(remaining < len) {
867  // Does not fit in our buffer. Forward immediately.
868  if(m_buffer.empty()) {
869  base::encode(buffer_, remaining, false);
870  } else {
871  m_buffer.append(buffer_, remaining);
872  base::encode(m_buffer.data(), m_buffer.size(), false);
873  m_buffer.clear();
874  }
875  buffer_ += remaining;
876  len -= remaining;
877  remaining = m_size;
878  }
879 
880  if(last || len == remaining) {
881  if(m_buffer.empty()) {
882  // Pass through immediately.
883  base::encode(buffer_, len, last);
884  } else {
885  // Got already something in the buffer. Concat and forward.
886  m_buffer.append(buffer_, len);
887  base::encode(m_buffer.data(), m_buffer.size(), last);
888  m_buffer.clear();
889  }
890  } else {
891  // Save for later.
892  m_buffer.append(buffer_, len);
893  }
894 }
