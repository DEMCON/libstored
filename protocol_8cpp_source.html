67 void AsciiEscapeLayer::decode(void* buffer, size_t len) {
68  char* p = static_cast<char*>(buffer);
69 
70  // Common case: there is no escape character.
71  size_t i = 0;
72  for(; i + 1 < len; i++)
73  if(unlikely(p[i] == Esc))
74  goto first_escape;
75 
76  // No escape characters.
77  base::decode(buffer, len);
78  return;
79 
80 first_escape:
81  // Process escape sequences in-place.
82  size_t decodeOffset = i;
83 
84 escape:
85  if(p[++i] == Esc)
86  p[decodeOffset] = (uint8_t)Esc;
87  else
88  p[decodeOffset] = (uint8_t)p[i] & (uint8_t)EscMask;
89  decodeOffset++;
90  i++;
91 
92  for(; i + 1 < len; i++, decodeOffset++) {
93  if(unlikely(p[i] == Esc))
94  goto escape;
95  else
96  p[decodeOffset] = p[i];
97  }
98 
99  p[decodeOffset++] = p[i];
100  base::decode(p, decodeOffset);
101 }

103 static char needEscape(char c) {
104  if(!((uint8_t)c & (uint8_t)~(uint8_t)AsciiEscapeLayer::EscMask)) {
105  switch(c) {
106  case '\t':
107  case '\n':
108  // Don't escape.
109  return 0;
110 
111  case '\r':
112  // Do escape \r, as Windows may inject it automatically. So, if \r
113  // is meant to be sent, escape it, such that the client may remove
114  // all (unescaped) \r's automatically.
115  default:
116  return (char)((uint8_t)c | 0x40u);
117  }
118  } else if(c == AsciiEscapeLayer::Esc) {
119  return c;
120  } else {
121  return 0;
122  }
123 }

125 void AsciiEscapeLayer::encode(void const* buffer, size_t len, bool last) {
126  uint8_t const* p = static_cast<uint8_t const*>(buffer);
127  uint8_t const* chunk = p;
128 
129  for(size_t i = 0; i < len; i++) {
130  char escaped = needEscape((char)p[i]);
131  if(unlikely(escaped)) {
132  // This is a to-be-escaped byte.
133  if(chunk < p + i)
134  base::encode(chunk, (size_t)(p + i - chunk), false);
135 
136  uint8_t const esc[2] = { AsciiEscapeLayer::Esc, (uint8_t)escaped };
137  base::encode(esc, sizeof(esc), last && i + 1 == len);
138  chunk = p + i + 1;
139  }
140  }
141 
142  if(likely(chunk < p + len) || (len == 0 && last))
143  base::encode(chunk, (size_t)(p + len - chunk), last);
144 }

177 void TerminalLayer::decode(void* buffer, size_t len) {
178  size_t nonDebugOffset = m_decodeState < StateDebug ? 0 : len;
179 
180  for(size_t i = 0; i < len; i++) {
181  char c = (static_cast<char*>(buffer))[i];
182 
183  switch(m_decodeState) {
184  case StateNormal:
185  if(unlikely(c == Esc))
186  m_decodeState = StateNormalEsc;
187  break;
188  case StateNormalEsc:
189  if(likely(c == EscStart)) {
190  nonDebugDecode(static_cast<char*>(buffer) + nonDebugOffset, i - nonDebugOffset - 1); // Also skip the ESC
191  m_decodeState = StateDebug;
192  nonDebugOffset = len;
193  } else
194  m_decodeState = StateNormal;
195  break;
196  case StateDebug:
197  if(unlikely(c == Esc))
198  m_decodeState = StateDebugEsc;
199  else
200  m_buffer.push_back(c);
201  break;
202  case StateDebugEsc:
203  if(likely(c == EscEnd)) {
204  base::decode(&m_buffer[0], m_buffer.size());
205  m_decodeState = StateNormal;
206  m_buffer.clear();
207  nonDebugOffset = i + 1;
208  } else {
209  m_decodeState = StateDebug;
210  m_buffer.push_back((char)Esc);
211  m_buffer.push_back(c);
212  }
213  break;
214  }
215  }
216 
217  if(nonDebugOffset < len)
218  nonDebugDecode(static_cast<char*>(buffer) + nonDebugOffset, len - nonDebugOffset);
219 }

323 void SegmentationLayer::decode(void* buffer, size_t len) {
324  if(len == 0)
325  return;
326 
327  char* buffer_ = static_cast<char*>(buffer);
328  if(!m_decode.empty() || buffer_[len - 1] != EndMarker) {
329  // Save for later packet reassembling.
330  size_t start = m_decode.size();
331  m_decode.resize(start + len);
332  memcpy(&m_decode[start], buffer_, len);
333 
334  size_t size = m_decode.size();
335  if(m_decode[size - 1] == EndMarker) {
336  // Got it.
337  base::decode(&m_decode[0], size - 1);
338  m_decode.clear();
339  }
340  } else {
341  // Full packet is in buffer. Forward immediately.
342  base::decode(buffer, len - 1);
343  }
344 }

346 void SegmentationLayer::encode(void const* buffer, size_t len, bool last) {
347  char const* buffer_ = static_cast<char const*>(buffer);
348 
349  size_t mtu = lowerMtu();
350  while(len) {
351  size_t remaining = mtu - m_encoded;
352  size_t chunk = std::min(len, remaining);
353  base::encode(buffer_, chunk, chunk == remaining);
354  len -= chunk;
355  buffer_ += chunk;
356 
357  if(chunk == remaining) {
358  // Full MTU.
359  m_encoded = 0;
360  } else {
361  // Partial MTU. Record that we already filled some of the packet.
362  m_encoded += chunk;
363  }
364  }
365 
366  if(last) {
367  // The marker always ends the packet.
368  char end = EndMarker;
369  base::encode(&end, 1, true);
370  m_encoded = 0;
371  }
372 }

397 void ArqLayer::decode(void* buffer, size_t len) {
398  if(len == 0)
399  return;
400 
401  char* buffer_ = static_cast<char*>(buffer);
402  uint8_t seq = (uint8_t)buffer_[0];
403  if(unlikely(seq == 0)) {
404  // Reset communication state.
405  m_decodeState = DecodeStateIdle;
406  m_encodeState = EncodeStateIdle;
407  m_encodeBuffer.clear();
408  m_encodeBufferSize = 0;
409  m_decodeSeq = 0;
410  m_encodeSeq = 0;
411  setPurgeableResponse(false);
412  }
413 
414  switch(m_decodeState) {
415  case DecodeStateDecoded:
416  stored_assert(m_encodeState == EncodeStateIdle || m_encodeState == EncodeStateUnbufferedIdle);
417 
418  if(seq == m_decodeSeq) {
419  // This is the next command. The previous one was received, apparently.
420  m_decodeState = DecodeStateIdle;
421  m_encodeState = EncodeStateIdle;
422  m_encodeBuffer.clear();
423  m_encodeBufferSize = 0;
424  setPurgeableResponse(false);
425  } else if(seq == m_decodeSeqStart) {
426  // This seems to be a retransmit of the current command.
427  switch(m_encodeState) {
428  case EncodeStateUnbufferedIdle:
429  // We must reexecute the command. Reset the sequence number, as the content may change.
430  setPurgeableResponse(false);
431  m_decodeSeq = m_decodeSeqStart;
432  m_decodeState = DecodeStateIdle;
433  break;
434  case EncodeStateIdle:
435  // Wait for full retransmit of the command, but do not actually decode.
436  m_decodeState = DecodeStateRetransmit;
437  break;
438  default:
439  stored_assert(false); // NOLINT(hicpp-static-assert,misc-static-assert)
440  }
441  } // else: unexpected seq; ignore.
442  break;
443  default:;
444  }
445 
446  switch(m_decodeState) {
447  case DecodeStateRetransmit: {
448  uint8_t nextSeq = (uint8_t)(seq + 1u);
449  if(nextSeq == 0)
450  nextSeq = 1;
451  if(nextSeq == m_decodeSeq) {
452  // Got the last part of the command. Retransmit the response buffer.
453  for(std::vector<std::string>::const_iterator it = m_encodeBuffer.cbegin(); it != m_encodeBuffer.cend(); ++it)
454  base::encode(it->data(), it->size(), true);
455  m_decodeState = DecodeStateDecoded;
456  }
457  break; }
458  case DecodeStateIdle:
459  if(unlikely(len == 1)) {
460  // Ignore empty packets.
461  if(++m_decodeSeq == 0)
462  m_decodeSeq = 1;
463  break;
464  }
465  m_decodeSeqStart = m_decodeSeq;
466  m_decodeState = DecodeStateDecoding;
467  // fall-through
468  case DecodeStateDecoding:
469  if(likely(seq == m_decodeSeq)) {
470  // Properly in sequence.
471  base::decode(buffer_ + 1, len - 1);
472  if(++m_decodeSeq == 0)
473  m_decodeSeq = 1;
474 
475  // Should not wrap-around during a request.
476  stored_assert(m_decodeSeq != m_decodeSeqStart);
477  }
478  break;
479  default:;
480  }
481 }

483 void ArqLayer::encode(void const* buffer, size_t len, bool last) {
484  if(m_decodeState == DecodeStateDecoding) {
485  // This seems to be the first part of the response.
486  // So, the request message must have been complete.
487  m_decodeState = DecodeStateDecoded;
488  stored_assert(m_encodeState == EncodeStateIdle || m_encodeState == EncodeStateUnbufferedIdle);
489  m_encodeSeq = 0;
490  }
491 
492  switch(m_encodeState) {
493  case EncodeStateIdle:
494  case EncodeStateEncoding:
495  if(unlikely(m_maxEncodeBuffer > 0 && m_encodeBufferSize + len > m_maxEncodeBuffer)) {
496  // Overflow.
497  setPurgeableResponse();
498  }
499  break;
500  default:;
501  }
502 
503  switch(m_encodeState) {
504  case EncodeStateUnbufferedIdle:
505  base::encode((void const*)&m_encodeSeq, 1, false);
506  if(++m_encodeSeq == 0)
507  m_encodeSeq = 1;
508  m_encodeState = EncodeStateUnbufferedEncoding;
509  // fall-through
510  case EncodeStateUnbufferedEncoding:
511  base::encode(buffer, len, last);
512  if(last)
513  m_encodeState = EncodeStateUnbufferedIdle;
514  break;
515 
516  case EncodeStateIdle:
517  {
518  char encodeSeq = (char)m_encodeSeq;
519 #if STORED_cplusplus >= 201103L
520  m_encodeBuffer.emplace_back(&encodeSeq, 1u);
521 #else
522  m_encodeBuffer.push_back(std::string(&encodeSeq, 1u));
523 #endif
524  }
525  if(++m_encodeSeq == 0)
526  m_encodeSeq = 1;
527  m_encodeState = EncodeStateEncoding;
528  m_encodeBufferSize++;
529  // fall-through
530  case EncodeStateEncoding:
531  stored_assert(!m_encodeBuffer.empty());
532  m_encodeBufferSize += len;
533  m_encodeBuffer.back().append(static_cast<char const*>(buffer), len);
534 
535  if(last) {
536  base::encode(m_encodeBuffer.back().data(), m_encodeBuffer.back().size(), true);
537  m_encodeState = EncodeStateIdle;
538  }
539  }
540 }

542 void ArqLayer::setPurgeableResponse(bool purgeable) {
543  bool wasPurgeable = m_encodeState == EncodeStateUnbufferedIdle || m_encodeState == EncodeStateUnbufferedEncoding;
544 
545  if(wasPurgeable == purgeable)
546  return;
547 
548  if(purgeable) {
549  // Switch to purgeable.
550  base::setPurgeableResponse(true);
551 
552  switch(m_encodeState) {
553  case EncodeStateEncoding: {
554  std::string const& s = m_encodeBuffer.back();
555  base::encode(s.data(), s.size(), false);
556  m_encodeState = EncodeStateUnbufferedEncoding;
557  break; }
558  case EncodeStateIdle:
559  m_encodeState = EncodeStateUnbufferedIdle;
560  break;
561  default:
562  stored_assert(false); // NOLINT(hicpp-static-assert,misc-static-assert)
563  }
564 
565  m_encodeBuffer.clear();
566  m_encodeBufferSize = 0;
567  } else {
568  // Switch to precious.
569  switch(m_encodeState) {
570  case EncodeStateUnbufferedEncoding:
571  // First part already sent. Can't switch to precious right now.
572  // Wait till start of next command.
573  break;
574  case EncodeStateUnbufferedIdle:
575  base::setPurgeableResponse(false);
576  m_encodeState = EncodeStateIdle;
577  break;
578  default:
579  stored_assert(false); // NOLINT(hicpp-static-assert,misc-static-assert)
580  }
581  }
