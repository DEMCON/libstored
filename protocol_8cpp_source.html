65 void AsciiEscapeLayer::decode(void* buffer, size_t len) {
66  char* p = static_cast<char*>(buffer);
67 
68  // Common case: there is no escape character.
69  size_t i = 0;
70  for(; i + 1 < len; i++)
71  if(unlikely(p[i] == Esc))
72  goto first_escape;
73 
74  // No escape characters.
75  base::decode(buffer, len);
76  return;
77 
78 first_escape:
79  // Process escape sequences in-place.
80  size_t decodeOffset = i;
81 
82 escape:
83  if(p[++i] == Esc)
84  p[decodeOffset] = (uint8_t)Esc;
85  else
86  p[decodeOffset] = (uint8_t)p[i] & (uint8_t)EscMask;
87  decodeOffset++;
88  i++;
89 
90  for(; i + 1 < len; i++, decodeOffset++) {
91  if(unlikely(p[i] == Esc))
92  goto escape;
93  else
94  p[decodeOffset] = p[i];
95  }
96 
97  p[decodeOffset++] = p[i];
98  base::decode(p, decodeOffset);
99 }

101 static char needEscape(char c) {
102  if(!((uint8_t)c & (uint8_t)~(uint8_t)AsciiEscapeLayer::EscMask)) {
103  switch(c) {
104  case '\t':
105  case '\n':
106  // Don't escape.
107  return 0;
108 
109  case '\r':
110  // Do escape \r, as Windows may inject it automatically. So, if \r
111  // is meant to be sent, escape it, such that the client may remove
112  // all (unescaped) \r's automatically.
113  default:
114  return (char)((uint8_t)c | 0x40u);
115  }
116  } else if(c == AsciiEscapeLayer::Esc) {
117  return c;
118  } else {
119  return 0;
120  }
121 }

127 template <typename void_type, typename char_type>
128 static void AsciiEscapeLayer_encode(AsciiEscapeLayer& that, void_type* buffer, size_t len, bool last) {
129  char_type* p = static_cast<char_type*>(buffer);
130  char_type* chunk = p;
131  for(size_t i = 0; i < len; i++) {
132  char escaped = needEscape((char)p[i]);
133  if(unlikely(escaped)) {
134  // This is a to-be-escaped byte.
135  if(chunk < p + i)
136  that.AsciiEscapeLayer::base::encode(chunk, (size_t)(p + i - chunk), false);
137 
138  char_type esc[2] = { AsciiEscapeLayer::Esc, (char_type)escaped };
139  that.AsciiEscapeLayer::base::encode(esc, sizeof(esc), last && i + 1 == len);
140  chunk = p + i + 1;
141  }
142  }
143 
144  if(likely(chunk < p + len) || (len == 0 && last))
145  that.AsciiEscapeLayer::base::encode(chunk, (size_t)(p + len - chunk), last);
146 }

182 void TerminalLayer::decode(void* buffer, size_t len) {
183  size_t nonDebugOffset = m_decodeState < StateDebug ? 0 : len;
184 
185  for(size_t i = 0; i < len; i++) {
186  char c = (static_cast<char*>(buffer))[i];
187 
188  switch(m_decodeState) {
189  case StateNormal:
190  if(unlikely(c == Esc))
191  m_decodeState = StateNormalEsc;
192  break;
193  case StateNormalEsc:
194  if(likely(c == EscStart)) {
195  nonDebugDecode(static_cast<char*>(buffer) + nonDebugOffset, i - nonDebugOffset - 1); // Also skip the ESC
196  m_decodeState = StateDebug;
197  nonDebugOffset = len;
198  } else
199  m_decodeState = StateNormal;
200  break;
201  case StateDebug:
202  if(unlikely(c == Esc))
203  m_decodeState = StateDebugEsc;
204  else
205  m_buffer.push_back(c);
206  break;
207  case StateDebugEsc:
208  if(likely(c == EscEnd)) {
209  base::decode(&m_buffer[0], m_buffer.size());
210  m_decodeState = StateNormal;
211  m_buffer.clear();
212  nonDebugOffset = i + 1;
213  } else {
214  m_decodeState = StateDebug;
215  m_buffer.push_back((char)Esc);
216  m_buffer.push_back(c);
217  }
218  break;
219  }
220  }
221 
222  if(nonDebugOffset < len)
223  nonDebugDecode(static_cast<char*>(buffer) + nonDebugOffset, len - nonDebugOffset);
224 }
