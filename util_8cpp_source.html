49 int strncmp(char const* __restrict__ str1, size_t len1, char const* __restrict__ str2, size_t len2) {
50  stored_assert(str1);
51  stored_assert(str2);
52 
53  size_t i = 0;
54  for(; i < len1 && i < len2; i++) {
55  char c1 = str1[i];
56  char c2 = str2[i];
57 
58  if(c1 < c2)
59  return -1;
60  else if(c1 > c2)
61  return 1;
62  else if(c1 == 0)
63  // early-terminate same strings
64  return 0;
65  }
66 
67  if(len1 == len2)
68  // same strings
69  return 0;
70  else if(i == len1)
71  // str1 was shortest
72  return str2[i] ? -1 : 0;
73  else
74  // str2 was shortest
75  return str1[i] ? 1 : 0;
76 }
