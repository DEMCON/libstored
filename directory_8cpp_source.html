30  T v = 0;
31  while(*p & 0x80u) {
32  v = (v | (T)(*p & 0x7fu)) << 7u;

63 Variant<> find(void* buffer, uint8_t const* directory, char const* name, size_t len) {
64  if(unlikely(!directory || !name)) {
65 notfound:
66  return Variant<>();
67  }
68 
69  uint8_t const* p = directory;
70  while(true) {
71  bool nameEnd = !*name || len == 0;
72  if(*p == 0) {
73  // end
74  goto notfound;
75  } else if(*p >= 0x80u) {
76  // var
77  Type::type type = (Type::type)(*p++ ^ 0x80u);
78  size_t datalen = !Type::isFixed(type) ? decodeInt<size_t>(p) : Type::size(type);
79  size_t offset = decodeInt<size_t>(p);
80  if(Type::isFunction(type))
81  return Variant<>(type, (unsigned int)offset, datalen);
82  else
83  return Variant<>(type, static_cast<char*>(buffer) + offset, datalen);
84  } else if(*p <= 0x1f) {
85  // skip
86  if(nameEnd)
87  goto notfound;
88 
89  uint8_t skip;
90  for(skip = *p++; skip > 0 && len > 0 && *name; skip--, name++, len--) {
91  switch(*name) {
92  case '/':
93  goto notfound;
94  default:;
95  }
96  }
97 
98  if(skip > 0)
99  // Premature end of name
100  goto notfound;
101  } else if(*p == '/') {
102  // Skip till next /
103  if(nameEnd)
104  goto notfound;
105 
106  while(len-- > 0 && *name++ != '/')
107  if(!*name)
108  goto notfound;
109  p++;
110  } else {
111  // match char
112  int c = (nameEnd ? 0 : (int)*name) - (int)*p++;
113  if(c < 0) {
114  // take jmp_l
115  p += decodeInt<uintptr_t>(p) - 1;
116  } else {
117  skipOffset(p);
118  if(c > 0) {
119  // take jmp_g
120  p += decodeInt<uintptr_t>(p) - 1;
121  } else {
122  // equal
123  skipOffset(p);
124  name++;
125  len--;
126  }
127  }
128  }
129  }
130 }

138 static void list(void* container, void* buffer, uint8_t const* directory, ListCallbackArg* f, void* arg, std::string& name)
139 {
140  if(unlikely(!buffer || !directory))
141  return;
142 
143  uint8_t const* p = directory;
144  size_t erase = 0;
145 
146  while(true) {
147  if(*p == 0) {
148  // end
149  break;
150  } else if(*p >= 0x80) {
151  // var
152  Type::type type = (Type::type)(*p++ ^ 0x80u);
153  size_t len = !Type::isFixed(type) ? decodeInt<size_t>(p) : Type::size(type);
154  size_t offset = decodeInt<size_t>(p);
155  char* b = Type::isFunction(type) ? nullptr : static_cast<char*>(buffer);
156  f(container, name.c_str(), type, b + offset, len, arg);
157  break;
158  } else if(*p <= 0x1f) {
159  // skip
160  name.append(*p, '?');
161  erase += *p;
162  p++;
163  } else if(*p == '/') {
164  // Hierarchy separator.
165  name.push_back('/');
166  erase++;
167  p++;
168  } else {
169  // next char in name
170  char c = (char)*p++;
171 
172  // take jmp_l
173  uintptr_t jmp = decodeInt<uintptr_t>(p);
174  list(container, buffer, p + jmp - 1, f, arg, name);
175 
176  // take jmp_g
177  jmp = decodeInt<uintptr_t>(p);
178  list(container, buffer, p + jmp - 1, f, arg, name);
179 
180  // resume with this char
181  name.push_back(c);
182  erase++;
183  }
184  }
185 
186  if(erase)
187  name.erase(name.end() - (long)erase, name.end());
188 }
