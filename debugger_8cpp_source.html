91 static bool checkPrefix(char const* name, char const* prefix, size_t len) {
92  stored_assert(name && prefix);
93  // Check full prefix, as we only check against prefix and not to others,
94  // so we cannot check disambiguity.
95  return memcmp(name, prefix, std::min(len, strlen(prefix))) == 0;

99 DebugVariant Debugger::find(char const* name, size_t len) const {
100  if(unlikely(!name)) {
101 notfound:
102  return DebugVariant();
103  }
104 
105  if(Config::DebuggerAlias > 0 && len == 1 && name[0] != '/') {
106  // This is probably an alias.
107  AliasMap::const_iterator it = aliases().find(name[0]);
108  if(it != aliases().end())
109  // Got it.
110  return it->second;
111  }
112 
113  if(m_map.size() == 1) {
114  // Don't compare prefix, just forward to the only mapped store.
115  return m_map.begin()->second->find(name, len);
116  }
117 
118  // name contains '/prefix/object', where '/prefix' equals he mapped name of
119  // a store.
120 
121  StoreMap::const_iterator it = m_map.upper_bound(name);
122  // If found, it->first:
123  // - is alphabetically after name, such as /prefix, while name was /pref/object (match)
124  // - is alphabetically after name, such as /zz, while name was /pref/object (no match)
125  // If not found, it equals end(), but (--it)->first:
126  // - is alphabetically before name, such as /prefix, while name was /prefix/object (match)
127  // - is alphabetically before name, such as /aa, while name was /prefix/object (no match)
128  // So, check both it and --it.
129 
130  if( it != m_map.end() && checkPrefix(name, it->first, len))
131  goto gotit;
132  else if(--it != m_map.end() && checkPrefix(name, it->first, len))
133  goto gotit;
134  else
135  goto notfound;
136 
137 gotit:
138  // Strip first / from name, and let the store eat the rest of the prefix.
139  return it->second->find(&name[1], len - 1);
140 }

147 void Debugger::map(DebugStoreBase* store, char const* name) {
148  if(!name && store)
149  name = store->name();
150 
151  if(!name || name[0] != '/' || !store)
152  return;
153 
154  StoreMap::iterator it = m_map.find(name);
155 
156  if(it == m_map.end()) {
157  // New; insert.
158 #if STORED_cplusplus >= 201103L
159  m_map.insert(StoreMap::value_type(name, std::unique_ptr<DebugStoreBase>(store)));
160 #else
161  m_map.insert(StoreMap::value_type(name, store));
162 #endif
163  } else {
164  // Replace previous mapping.
165 #if STORED_cplusplus >= 201103L
166  it->second.reset(store);
167 #else
168  cleanup(it->second);
169  it->second = store;
170 #endif
171  }

353 void Debugger::process(void const* frame, size_t len, ProtocolLayer& response) {
354  if(unlikely(!frame || len == 0))
355  return;
356 
357  ScratchPad<>::Snapshot snapshot = spm().snapshot();
358 
359  char const* p = static_cast<char const*>(frame);
360 
361  switch(p[0]) {
362  case CmdCapabilities: {
363  // get capabilities
364  // Request: '?'
365  // Response: <list of command chars>
366 
367  char* c;
368  size_t cs;
369  capabilities(c, cs);
370  response.encode(c, cs, true);
371  return;
372  }
373  case CmdRead: {
374  // read an object
375  // Request: 'r' </path/to/object>
376  // Response: <hex-encoded value>
377 
378  if(!Config::DebuggerRead)
379  goto error;
380 
381  DebugVariant v = find(++p, --len);
382  if(unlikely(!v.valid()))
383  goto error;
384 
385  size_t size = v.size();
386  void* data = spm().alloc<char>(size);
387  size = v.get(data, size);
388  encodeHex(v.type(), data, size);
389  response.encode(data, size, true);
390  return;
391  }
392  case CmdWrite: {
393  // write an object
394  // Request: 'w' <hex-encoded value> </path/to/object>
395  // Response: '?' | '!'
396 
397  if(!Config::DebuggerWrite)
398  goto error;
399 
400  void const* value = ++p;
401  len--;
402  // Find / that indicates the object.
403  // Always use last char, which may be an alias.
404  while(len > 1 && *p != '/') { p++; len--; }
405  if(len == 0)
406  goto error;
407 
408  size_t valuelen = (size_t)(static_cast<char const*>(p) - static_cast<char const*>(value));
409  DebugVariant variant = find(p, len);
410  if(!variant.valid())
411  goto error;
412 
413  if(!decodeHex(variant.type(), value, valuelen))
414  goto error;
415 
416  variant.set(value, valuelen);
417  break;
418  }
419  case CmdEcho: {
420  // echo the request
421  // Request: 'e' <any data>
422  // Response: <the data>
423 
424  if(!Config::DebuggerEcho)
425  goto error;
426 
427  response.encode(p + 1, len - 1, true);
428  return;
429  }
430  case CmdList: {
431  // list all objects
432  // Request: 'l'
433  // Response: ( <type byte in hex> <length in hex> <name of object> '\n' ) *
434 
435  if(!Config::DebuggerList)
436  goto error;
437 
438  ListCmdCallbackArg arg = {this, &response, 0};
439 
440  list(&listCmdCallback, &arg);
441 
442  if(!arg.callbacks)
443  goto error;
444 
445  response.encode();
446  return;
447  }
448  case CmdAlias: {
449  // define an alias
450  // Request: 'a' <char> </path/to/object> ?
451  // Response: '!' | '?'
452 
453  if(Config::DebuggerAlias <= 0)
454  goto error;
455 
456  if(len < 2)
457  goto error;
458 
459  char a = p[1];
460  if(a < 0x20 || a > 0x7e || a == '/')
461  // invalid
462  goto error;
463 
464  if(len == 2) {
465  // erase given alias
466  aliases().erase(a);
467  break;
468  }
469 
470  DebugVariant variant = find(&p[2], len - 2);
471  if(!variant.valid())
472  goto error;
473 
474  if(aliases().size() == Config::DebuggerAlias) {
475  // Only accept this alias if it replaces one.
476  AliasMap::iterator it = aliases().find(a);
477  if(it == aliases().end())
478  // Overflow.
479  goto error;
480 
481  // Replace.
482  it->second = variant;
483  } else {
484  // Add new alias.
485  std::pair<AliasMap::iterator, bool> inserted = aliases().insert(std::make_pair(a, variant));
486  if(!inserted.second)
487  // Already there. Replace it.
488  inserted.first->second = variant;
489  }
490  break;
491  }
492  case CmdMacro: {
493  // define a macro
494  // Request: 'm' <char> ( <sep> <any command> ) *
495  // Response: '!' | '?'
496 
497  if(Config::DebuggerMacro <= 0)
498  goto error;
499 
500  if(len < 2)
501  goto error;
502 
503  char m = p[1];
504  if(len == 2) {
505  // Erase macro
506  MacroMap::iterator it = macros().find(m);
507  if(it != macros().end()) {
508  stored_assert(it->second.size() <= m_macroSize);
509  m_macroSize -= it->second.size();
510  macros().erase(it);
511  }
512  break;
513  }
514 
515  p += 2;
516  len -= 2;
517 
518  MacroMap::iterator it = macros().find(m);
519  if(it == macros().end()) {
520  // New macro.
521  size_t newlen = len + m_macroSize;
522  if(newlen > Config::DebuggerMacro)
523  goto error;
524 
525  macros().insert(MacroMap::value_type(m, std::string(p, len)));
526  m_macroSize = newlen;
527  } else {
528  // Update existing macro.
529  size_t newlen = len + m_macroSize - it->second.size();;
530  if(newlen > Config::DebuggerMacro)
531  goto error;
532 
533  it->second = std::string(p, len);
534  m_macroSize = newlen;
535  }
536  break;
537  }
538  case CmdIdentification: {
539  // Return identification string
540  // Request: 'i'
541  // Response: <UTF-8 encoded string>
542 
543  if(!Config::DebuggerIdentification)
544  goto error;
545 
546  char const* id = identification();
547  if(!id || !*id)
548  // Not supported, apparently.
549  goto error;
550 
551  response.encode(id, strlen(id), true);
552  return;
553  }
554  case CmdVersion: {
555  // Return version(s)
556  // Request: 'v'
557  // Response: <debugger version> ( ' ' <application-defined version> ) *
558 
559  if(!Config::DebuggerVersion)
560  goto error;
561 
562  if(!version(response))
563  // No version output.
564  goto error;
565 
566  response.encode();
567  return;
568  }
569  case CmdReadMem: {
570  // Read memory directly.
571  // Request: 'R' <ASCII-hex address> ( ' ' <length> ) ?
572  // Response: '?' | <ASCII-hex bytes>
573 
574  if(!Config::DebuggerReadMem)
575  goto error;
576 
577  if(len < 2)
578  goto error;
579 
580  char const* addrhex = ++p;
581 
582  // Find end of address
583  size_t bufferlen = 1;
584  p++;
585  len -= 2;
586  for(; len > 0 && *p != ' '; bufferlen++, p++, len--);
587 
588  void const* buffer = addrhex;
589  if(!decodeHex(Type::Pointer, buffer, bufferlen))
590  goto error;
591 
592  stored_assert(bufferlen == sizeof(void*));
593  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
594  char* addr = *reinterpret_cast<char* const*>(buffer);
595  size_t datalen = sizeof(void*);
596 
597  if(len > 1) {
598  buffer = p + 1;
599  bufferlen = len - 1;
600  if(!decodeHex(Type::Uint, buffer, bufferlen))
601  goto error;
602 
603  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
604  datalen = (size_t)*reinterpret_cast<unsigned int const*>(buffer);
605  }
606 
607  if(datalen == 0)
608  goto error;
609 
610  // Go read the memory in chunks
611  while(datalen > 0) {
612  ScratchPad<>::Snapshot chunk_snapshot = spm().snapshot();
613 
614  size_t chunk = std::min<size_t>(64u, datalen);
615  void* r = addr;
616  size_t rlen = chunk;
617  encodeHex(Type::Blob, r, rlen);
618  response.encode(r, rlen, false);
619  addr += chunk;
620  datalen -= chunk;
621  }
622 
623  response.encode();
624  return;
625  }
626  case CmdWriteMem: {
627  // Write memory directly.
628  // Request: 'W' <ASCII-hex address> ' ' <ASCII-hex bytes>
629  // Response: '!' | '?'
630  //
631  if(!Config::DebuggerWriteMem)
632  goto error;
633 
634  if(len < 2)
635  goto error;
636 
637  char const* addrhex = ++p;
638 
639  // Find length of address
640  size_t bufferlen = 1;
641  p++;
642  len -= 2;
643  for(; len > 0 && *p != ' '; bufferlen++, p++, len--);
644 
645  void const* buffer = addrhex;
646  if(!decodeHex(Type::Pointer, buffer, bufferlen))
647  goto error;
648 
649  stored_assert(bufferlen == sizeof(void*));
650  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
651  char* addr = *reinterpret_cast<char* const*>(buffer);
652 
653  // p should point to the ' ' before the data
654  if(len < 3) // ' ' and two hex chars of the first byte
655  goto error;
656 
657  len--;
658  p++;
659 
660  if(len & 1u)
661  // Odd number of nibbles.
662  goto error;
663 
664  // Process data in chunks.
665  while(len > 0) {
666  ScratchPad<>::Snapshot chunk_snapshot = spm().snapshot();
667 
668  size_t chunk = std::min<size_t>(64u, len);
669  void const* w = p;
670  size_t wlen = chunk;
671  if(!decodeHex(Type::Blob, w, wlen))
672  goto error;
673  memcpy(addr, w, wlen);
674  addr += chunk;
675  p += chunk;
676  len -= chunk;
677  }
678  break;
679  }
680  case CmdStream: {
681  // Get stream data
682  // Request: 's' ( <stream char> <suffix> ? ) ?
683  //
684  // Without stream char:
685  // Response: '?' | <stream char> +
686  //
687  // With stream char
688  // Response: <stream data> <suffix>
689 
690  if(Config::DebuggerStreams < 1)
691  goto error;
692 
693  if(len == 1) {
694  void const* buffer;
695  size_t bufferlen;
696  streams(buffer, bufferlen);
697  if(bufferlen == 0)
698  goto error;
699 
700  response.encode(buffer, bufferlen);
701  return;
702  }
703 
704  char s = p[1];
705  char const* suffix = &p[2];
706  size_t suffixlen = len - 2;
707 
708  std::string* str = stream(s);
709  if(!str)
710  goto error;
711 
712  response.encode(str->data(), str->size(), false);
713  str->clear();
714 
715  response.encode(suffix, suffixlen);
716  return;
717  }
718  case CmdTrace: {
719  // Configure trace
720  //
721  // Enable:
722  // Request: 't' <macro> <stream> ( <decimate in hex> ) ?
723  //
724  // Disable:
725  // Request: 't'
726  //
727  // Response: `?` | `!`
728 
729  if(!Config::DebuggerTrace)
730  goto error;
731 
732  // Always disable by default.
733  m_traceDecimate = 0;
734 
735  if(len == 1)
736  // Disable tracing.
737  break;
738 
739  // Enable tracing
740  if(len < 3)
741  goto error;
742 
743  m_traceMacro = p[1];
744  m_traceStream = p[2];
745 
746  if(len > 3) {
747  void const* d = &p[3];
748  size_t dlen = std::min<size_t>(8, len - 3); // at most 32-bit
749  if(!decodeHex(Type::Uint, d, dlen))
750  goto error;
751 
752  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
753  m_traceDecimate = *reinterpret_cast<unsigned int const*>(d);
754  } else
755  m_traceDecimate = 1;
756  break;
757  }
758  default:
759  // Unknown command.
760 
761  if(Config::DebuggerMacro && !macros().empty()) {
762  // Try parsing this as a macro.
763  if(runMacro(p[0], response))
764  return;
765  }
766 
767  goto error;
768  }
769 
770  {
771  char ack = Ack;
772  response.encode(&ack, 1, true);
773  }
774  return;
775 error:
776  char nack = Nack;
777  response.encode(&nack, 1, true);
778 }

804 bool Debugger::runMacro(char m, ProtocolLayer& response) {
805  MacroMap::iterator it = macros().find(m);
806 
807  if(it == macros().end())
808  // Unknown macro.
809  return false;
810 
811  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-const-cast)
812  std::string const& definition = const_cast<std::string const&>(it->second);
813 
814  // Expect the separator and at least one char to execute.
815  if(definition.size() < 2)
816  // Nothing to do.
817  return true;
818 
819  char sep = definition[0];
820 
821  FrameMerger merger(response);
822 
823  size_t pos = 0;
824  do {
825  size_t nextpos = definition.find(sep, ++pos);
826  size_t len;
827  if(nextpos == std::string::npos)
828  len = definition.size() - pos;
829  else
830  len = nextpos - pos;
831  process(&definition[pos], len, merger);
832  pos = nextpos;
833  } while(pos != std::string::npos);
834 
835  response.encode();
836  return true;
837 }

860 void Debugger::encodeHex(Type::type type, void*& data, size_t& len, bool shortest) {
861  if(len == 0)
862  return;
863 
864  if(!data) {
865  len = 0;
866  return;
867  }
868 
869  uint8_t const* src = static_cast<uint8_t const*>(data);
870 
871  if(shortest && type == Type::Bool) {
872  // single char
873  char* hex = spm().alloc<char>(2);
874  hex[0] = src[0] ? '1' : '0';
875  hex[1] = 0;
876  len = 1;
877  data = hex;
878  return;
879  }
880 
881  char* hex = spm().alloc<char>(len * 2 + 1);
882 
883  if(Type::isFixed(type)) {
884  // big endian
885  for(size_t i = 0; i < len; i++, src++) {
886 #ifdef STORED_LITTLE_ENDIAN
887  hex[(len - i) * 2 - 2] = encodeNibble((uint8_t)(*src >> 4u));
888  hex[(len - i) * 2 - 1] = encodeNibble(*src);
889 #else
890  hex[i * 2] = encodeNibble(*src >> 4u);
891  hex[i * 2 + 1] = encodeNibble(*src);
892 #endif
893  }
894 
895  len *= 2;
896 
897  if(shortest && Type::isInt(type))

1072 
1073  if(it != m_streams.end())
1074  return &it->second;
1075  else if(alloc) {
1076  if(m_streams.size() >= Config::DebuggerStreams) {
1077  // Out of buffers. Check if we can recycle something.
1078  bool cleaned = true;
1079  while(cleaned) {
1080  cleaned = false;
1081  for(StreamMap::iterator it2 = m_streams.begin(); it2 != m_streams.end(); ++it2)
1082  if(it2->second.empty()) {

1084  m_streams.erase(it2);
1085  cleaned = true;
1086  break;
1087  }
1088  }
1089  }
1090

1092  // Add a new stream.
1093  return &m_streams.insert(std::make_pair(s, std::string())).first->second;
1094  else
