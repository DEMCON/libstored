53  struct Type {
54  enum type {
55  MaskSize = 0x07u,
56  MaskFlags = 0x78u,
57  FlagSigned = 0x08u,
58  FlagInt = 0x10u,
59  FlagFixed = 0x20u,
60  FlagFunction = 0x40u,
61 
62  // int
63  Int8 = FlagFixed | FlagInt | FlagSigned | 0u,
64  Uint8 = FlagFixed | FlagInt | 0u,
65  Int16 = FlagFixed | FlagInt | FlagSigned | 1u,
66  Uint16 = FlagFixed | FlagInt | 1u,
67  Int32 = FlagFixed | FlagInt | FlagSigned | 3u,
68  Uint32 = FlagFixed | FlagInt | 3u,
69  Int64 = FlagFixed | FlagInt | FlagSigned | 7u,
70  Uint64 = FlagFixed | FlagInt | 7u,
71  Int = FlagFixed | FlagInt | (sizeof(int) - 1),
72  Uint = FlagFixed | (sizeof(int) - 1),
73 
74  // things with fixed length
75  Float = FlagFixed | FlagSigned | 3u,
76  Double = FlagFixed | FlagSigned | 7u,
77  Bool = FlagFixed | 0u,
78  Pointer32 = FlagFixed | 3u,
79  Pointer64 = FlagFixed | 7u,
80  Pointer = (sizeof(void*) <= 4 ? Pointer32 : Pointer64),
81 
82  // (special) things with undefined length
83  Void = 0u,
84  Blob = 1u,
85  String = 2u,
86 
87  Invalid = 0xffu,
88  };

91  static bool isFunction(type t) { return t & FlagFunction; }

95  static bool isInt(type t) { return isFixed(t) && (t & FlagInt); }
97  static bool isSigned(type t) { return isFixed(t) && (t & FlagSigned); }
99  static bool isSpecial(type t) { return (t & MaskFlags) == 0; }
101  static size_t size(type t) { return !isFixed(t) ? 0u : (size_t)(t & MaskSize) + 1u; }

736  size_t get(void* dst, size_t len = 0) const {
737  if(Type::isFixed(type())) {
738  stored_assert(len == size() || len == 0);
739  len = size();
740  } else {
741  len = std::min(len, size());
742  }
743 
744  if(Type::isFunction(type())) {
745  len = container().callback(false, dst, len, (unsigned int)m_f);
746  } else {
747  entryRO(len);
748  if(unlikely(type() == Type::String)) {
749  char* dst_ = static_cast<char*>(dst);
750  char* buffer_ = static_cast<char*>(m_buffer);
751  size_t len_ = strncpy(dst_, buffer_, len);
752  if(len > len_)
753  dst_[len_] = '\0';
754  len = len_;
755  } else {
756  memcpy(dst, m_buffer, len);
757  }
758  exitRO(len);
759  }
760  return len;
761  }
