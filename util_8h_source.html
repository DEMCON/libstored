217  template <typename R> struct saturated_cast_helper
218  {
219  template <typename T> __attribute__((pure)) static R cast(T value)
220  {
221  // Lower bound check
222  if(std::numeric_limits<R>::is_integer) {
223  if(!std::numeric_limits<T>::is_signed) {
224  // No need to check
225  } else if(!std::numeric_limits<R>::is_signed) {
226  if(value <= 0)
227  return 0;
228  } else {
229  // Both are signed.
230  if(static_cast<typename signedness_helper<T>::signed_type>(value) <= static_cast<typename signedness_helper<R>::signed_type>(std::numeric_limits<R>::min()))
231  return std::numeric_limits<R>::min();
232  }
233  } else {
234 #ifdef STORED_COMPILER_MSVC
235 # pragma warning( push )
236 # pragma warning( disable : 4146 ) // This code path is not triggered in case of unsigned ints.
237 #endif
238 
239 #pragma GCC diagnostic push
240 #pragma GCC diagnostic ignored "-Woverflow" // This error triggers when R is integer, but this code path is not triggered then.
241  if(value <= -std::numeric_limits<R>::max())
242  return -std::numeric_limits<R>::max();
243 #pragma GCC diagnostic pop
244 
245 #ifdef STORED_COMPILER_MSVC
246 # pragma warning( pop )
247 #endif
248  }
249 
250  // Upper bound check
251  if(value > 0)
252  if(static_cast<typename signedness_helper<T>::unsigned_type>(value) >= static_cast<typename signedness_helper<R>::unsigned_type>(std::numeric_limits<R>::max()))
253  return std::numeric_limits<R>::max();
254 
255  // Default conversion
256  return static_cast<R>(value);
257  }
