368  template <typename R> struct saturated_cast_helper
369  {
370  template <typename T> __attribute__((pure)) static R cast(T value)
371  {
372  // Lower bound check
373  if(std::numeric_limits<R>::is_integer) {
374  if(!std::numeric_limits<T>::is_signed) {
375  // No need to check
376  } else if(!std::numeric_limits<R>::is_signed) {
377  if(value <= 0)
378  return 0;
379  } else {
380  // Both are signed.
381  if(static_cast<typename signedness_helper<T>::signed_type>(value) <= static_cast<typename signedness_helper<R>::signed_type>(std::numeric_limits<R>::min()))
382  return std::numeric_limits<R>::min();
383  }
384  } else {
385 #ifdef STORED_COMPILER_MSVC
386 # pragma warning( push )
387 # pragma warning( disable : 4146 ) // This code path is not triggered in case of unsigned ints.
388 #endif
389 
390 #pragma GCC diagnostic push
391 #pragma GCC diagnostic ignored "-Woverflow" // This error triggers when R is integer, but this code path is not triggered then.
392  if(value <= -std::numeric_limits<R>::max())
393  return -std::numeric_limits<R>::max();
394 #pragma GCC diagnostic pop
395 
396 #ifdef STORED_COMPILER_MSVC
397 # pragma warning( pop )
398 #endif
399  }
400 
401  // Upper bound check
402  if(value > 0)
403  if(static_cast<typename signedness_helper<T>::unsigned_type>(value) >= static_cast<typename signedness_helper<R>::unsigned_type>(std::numeric_limits<R>::max()))
404  return std::numeric_limits<R>::max();
405 
406  // Default conversion
407  return static_cast<R>(value);
408  }
