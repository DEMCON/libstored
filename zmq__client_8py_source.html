231  def _decode(self, rep):
232  dtype = self._type & ~self.FlagFunction
233  try:
234  if self.isFixed():

236  if sys.version_info.major <= 3 and sys.version_info.minor < 9:

238  # See https://bugreports.qt.io/browse/PYSIDE-648

244  elif dtype == self.Int8:

760  def time(self):
761  if self._t == False:
762  # Not found
763  return None
764  elif self._t != None:
765  return self._t
766 
767  # Not initialized.
768  self._t = False
769 
770  if isinstance(self._t, str):
771  # Take the given time variable
772  t = self.find(self._t)
773  else:
774  # Try finding /t (unit)
775  t = self.find('/t (')
776 
777  if t == None:
778  # Not found, try the first /store/t (unit)
779  for o in self.list():
780  chunks = o.name.split('/', 4)
781  if len(chunks) != 3:
782  # Strange name
783  continue
784  elif chunks[2].name.startsWith('t ('):
785  # Got some
786  t = o
787  break;
788  if t == None:
789  # Still not found. Give up.
790  return None
791  elif isinstance(t, list):
792  # Multiple found. Just take first one.
793  t = t[0]
794  else:
795  # One found
796  pass
797 
798  try:
799  t0 = t.read()
