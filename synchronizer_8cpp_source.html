68 StoreJournal::Seq StoreJournal::bumpSeq() {
69  if(!m_partialSeq)
70  return m_seq;
71 
72  m_partialSeq = false;
73  m_seq++;
74 
75  Seq const saveRange = (1u << (sizeof(ShortSeq) * 8u)) - SeqLowerMargin;
76 
77  if(unlikely(m_seq - m_seqLower > saveRange)) {
78  Seq seqLower = m_seq;
79  m_seqLower = m_seq - SeqLowerMargin;
80 
81  for(size_t i = 0; i < m_changes.size(); i++) {
82  ObjectInfo& o = m_changes[i];
83  Seq o_seq = toLong(o.seq);
84  if(m_seq - o_seq > saveRange) {
85  update(o.key, o.len, m_seqLower, 0, m_changes.size());
86  seqLower = m_seqLower;
87  } else
88  seqLower = std::min(seqLower, o_seq);
89  }
90 
91  m_seqLower = seqLower;
92  }
93 
94  return m_seq;
95 }

102  ShortSeq sseq = toShort(m_seq);

126 bool StoreJournal::update(StoreJournal::Key key, size_t len, StoreJournal::Seq seq, size_t lower, size_t upper) {
127  if(lower >= upper)
128  return false;
129 
130  size_t pivot = (upper - lower) / 2 + lower;
131  ObjectInfo& o = m_changes[pivot];
132 
133  o.highest = toShort(std::max(toLong(o.highest), seq));
134 
135  if(o.key == key) {
136  o.seq = toShort(seq);
137  o.len = (Size)len;
138  return true;
139  } else if(key < o.key)
140  return update(key, len, seq, lower, pivot);
141  else
142  return update(key, len, seq, pivot + 1, upper);
143 }

170 bool StoreJournal::hasChanged(Key key, Seq since) const {
171  size_t lower = 0;
172  size_t upper = m_changes.size();
173 
174  while(lower < upper) {
175  size_t pivot = (upper - lower) / 2 + lower;
176 
177  ObjectInfo const& o = m_changes[pivot];
178 
179  if(toLong(o.highest) < since)
180  return false;
181  if(o.key == key)
182  return toLong(o.seq) >= since;
183 
184  if(o.key > key)
185  upper = pivot;
186  else
187  lower = pivot + 1;
188  }
189 
190  return false;
191 }

318 StoreJournal::Seq StoreJournal::decodeUpdates(void*& buffer, size_t& len) {
319  uint8_t* buffer_ = static_cast<uint8_t*>(buffer);
320  bool ok = true;
321 
322  while(len) {
323  Key key = decodeKey(buffer_, len, ok);
324  Size size = (Size)decodeKey(buffer_, len, ok);
325  void* obj = keyToBuffer(key, size, &ok);
326  if(!ok || len < size) {
327  buffer = buffer_;
328  return 0;
329  }
330 
331  memcpy(obj, buffer_, size);
332  buffer_ += size;
333  len -= size;
334  changed(key, size);
335  }
336 
337  return bumpSeq();
338 }

435 void SyncConnection::drop(StoreJournal& store) {
436  bool bye = false;
437 
438  Id id = 0;
439  for(IdInMap::iterator it = m_idIn.begin(); !id && it != m_idIn.end(); ++it)
440  if(it->second == &store)
441  id = it->first;
442 
443  if(id && m_idIn.erase(id))
444  bye = true;
445  if(m_idOut.erase(&store))
446  bye = true;
447  if(m_seq.erase(&store))

450  if(!bye)
451  return;

550 void SyncConnection::process(StoreJournal& store) {
551  SeqMap::iterator seq = m_seq.find(&store);
552  if(seq == m_seq.end())
553  // Unknown store.
554  return;
555  if(!store.hasChanged(seq->second))
556  // No recent changes.
557  return;
558 
559  IdOutMap::iterator id = m_idOut.find(&store);
560  if(id == m_idOut.end())
561  // No id related to this store.
562  return;
563 
564  encodeCmd(Update);
565  encodeId(id->second, false);
566  // Make sure to set the process seq before the last part of the encode is sent.
567  seq->second = store.encodeUpdates(*this, seq->second);
568  encode();
569 }

581 void SyncConnection::decode(void* buffer, size_t len) {
582  switch(decodeCmd(buffer, len)) {
583  case Hello: {
584  /*
585  * Hello
586  *
587  * 'h' hash id
588  */
589  char const* hash = StoreJournal::decodeHash(buffer, len);
590  Id id = decodeId(buffer, len);
591  if(!hash || !id)
592  break;
593 
594  StoreJournal* j = synchronizer().toJournal(hash);
595  if(!j) {
596  bye(hash);
597  break;
598  }
599 
600  encodeCmd(Welcome);
601  encodeId(id);
602 
603  m_idOut[j] = id;
604  id = nextId();
605  m_idIn[id] = j;
606 
607  encodeId(id, false);
608  m_seq[j] = j->encodeBuffer(*this, true);
609  break;
610  }
611  case Welcome: {
612  /*
613  * Welcome
614  *
615  * 'w' hello_id welcome_id buffer
616  */
617 
618  Id id = decodeId(buffer, len);
619  if(!id)
620  break;
621 
622  Id welcome_id = decodeId(buffer, len);
623  IdInMap::iterator j = m_idIn.find(id);
624 
625  StoreJournal::Seq seq = 0;
626 
627  if(!welcome_id || j == m_idIn.end() ||
628  !(seq = j->second->decodeBuffer(buffer, len)))
629  {
630  bye(id);
631  break;
632  }
633 
634  m_seq[j->second] = seq;
635  m_idOut[j->second] = welcome_id;
636  break;
637  }
638  case Update: {
639  /*
640  * Update
641  *
642  * 'u' id updates
643  */
644  Id id = decodeId(buffer, len);
645  if(!id)
646  return;
647 
648  IdInMap::iterator it = m_idIn.find(id);
649  if(it == m_idIn.end()) {
650  bye(id);
651  break;
652  }
653 
654  // Make sure that local changes are flushed out first.
655  process(*it->second);
656 
657  StoreJournal::Seq seq = 0;
658  if(!(seq = it->second->decodeUpdates(buffer, len))) {
659  bye(id);
660  break;
661  }
662 
663  m_seq[it->second] = seq;
664  break;
665  }
666  case Bye: {
667  /*
668  * Bye
669  *
670  * "I do not need any more updates of the given store (by hash, by id or all)."
671  *
672  * 'b' hash
673  * 'b' id
674  * 'b'
675  */
676  if(len == 0) {
677  m_seq.clear();
678  m_idIn.clear();
679  m_idOut.clear();
680  } else if(len == sizeof(Id)) {
681  Id id = decodeId(buffer, len);
682  if(!id)
683  break;
684 
685  IdInMap::iterator it = m_idIn.find(id);
686  if(it == m_idIn.end())
687  break;
688 
689  erase(id);
690  } else {
691  char const* hash = StoreJournal::decodeHash(buffer, len);
692  erase(hash);
693  }
694  break;
695  }
696  default:;
697  // Other; ignore.
698  }
699 }
