120 int ZmqLayer::recv(bool block) {
121  int res = 0;
122  int more;
123 
124  zmq_msg_t msg;
125 
126  if(unlikely(zmq_msg_init(&msg) == -1)) {
127  res = errno;
128  goto error_msg;
129  }
130 
131  if(unlikely(zmq_msg_recv(&msg, m_socket, block ? 0 : ZMQ_DONTWAIT) == -1)) {
132  res = errno;
133  if(res == EFSM) {
134  // We should not be receiving at the moment.
135  // That is odd, but we can try to recover from it by sending an (error) reply.
136  encode("?", 1);
137  }
138  goto error_recv;
139  }
140 
141  more = zmq_msg_more(&msg);
142 
143  if(unlikely(m_bufferSize || more || zmq_msg_get(&msg, ZMQ_SHARED))) {
144  // Save for later processing.
145  size_t msgSize = zmq_msg_size(&msg);
146  size_t newBufferSize = m_bufferSize + msgSize;
147 
148  if(newBufferSize > m_bufferCapacity) {
149  // NOLINTNEXTLINE(cppcoreguidelines-no-malloc,cppcoreguidelines-owning-memory)
150  void* p = realloc(m_buffer, newBufferSize);
151  if(!p) {
152  res = ENOMEM;
153  goto error_buffer;
154  }
155 
156  m_buffer = p;
157  m_bufferCapacity = newBufferSize;
158  }
159 
160  memcpy(static_cast<char*>(m_buffer) + m_bufferSize, zmq_msg_data(&msg), msgSize);
161  m_bufferSize = newBufferSize;
162  }
163 
164  if(likely(!more)) {
165  if(likely(!m_bufferSize)) {
166  // Process immediately, without copying to the buffer.
167  decode(zmq_msg_data(&msg), zmq_msg_size(&msg));
168  } else {
169  // Process message from buffer.
170  decode(m_buffer, m_bufferSize);
171  m_bufferSize = 0;
172  }
173  }
174 
175  zmq_msg_close(&msg);
176  return 0;
177 
178 error_buffer:
179 error_recv:
180  zmq_msg_close(&msg);
181 error_msg:
182  return res ? res : EIO;
183 }
