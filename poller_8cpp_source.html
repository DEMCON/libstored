52 static int setEvents(SOCKET s, HANDLE h, Poller::events_t events) {
53  stored_assert(s != INVALID_SOCKET); // NOLINT(hicpp-signed-bitwise)
54  stored_assert(h);
55 
56  long e = FD_CLOSE; // NOLINT(hicpp-signed-bitwise)
57  if(EVENTS_IS_SET(events, Poller::PollIn)) {
58  EVENTS_SET(e, FD_READ); // NOLINT(hicpp-signed-bitwise)
59  EVENTS_SET(e, FD_ACCEPT); // NOLINT(hicpp-signed-bitwise)
60  EVENTS_SET(e, FD_OOB); // NOLINT(hicpp-signed-bitwise)
61  }
62  if(EVENTS_IS_SET(events, Poller::PollOut))
63  EVENTS_SET(e, FD_WRITE); // NOLINT(hicpp-signed-bitwise)
64 
65  if(WSAEventSelect(s, h, e))
66  return EIO;
67 
68  return 0;
69 }

108 static int fd_to_HANDLE(int fd, HANDLE& h) {
109  intptr_t res = _get_osfhandle(fd);
110 
111  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-cstyle-cast)
112  if((HANDLE)res == INVALID_HANDLE_VALUE) {
113  return EINVAL;
114  } else if(res == -2) {
115  // stdin/stdout/stderr has no stream associated
116  // Always block.
117  if(!dummyEventReset)
118  if(!(dummyEventReset = CreateEvent(NULL, TRUE, FALSE, NULL)))
119  return EIO;
120 
121  h = dummyEventReset;
122  return 0;
123  } else if(fd == _fileno(stdout) || fd == _fileno(stderr)) {
124  // Cannot WaitFor... on stdout/stderr. Always return immediately using a dummy event.
125  if(!dummyEventSet)
126  if(!(dummyEventSet = CreateEvent(NULL, TRUE, TRUE, NULL)))
127  return EIO;
128 
129  h = dummyEventSet;
130  return 0;
131  } else {
132  h = (HANDLE)res; // NOLINT(cppcoreguidelines-pro-type-cstyle-cast)
133  return 0;
134  }
135 }

285 int Poller::add(Poller::Event const& e, void* user_data, events_t events) {
286 #ifdef STORED_POLL_WFMO
287  if(m_events.size() == MAXIMUM_WAIT_OBJECTS)
288  return EINVAL;
289 #endif
290 
291  for(std::deque<Event>::iterator it = m_events.begin(); it != m_events.end(); ++it)
292  if(*it == e)
293  return EINVAL;
294 
295  m_events.push_back(e);
296 
297  Event& b = m_events.back();
298  b.user_data = user_data;
299  b.events = events;
300 
301 #ifdef STORED_OS_WINDOWS
302  int res = 0;
303  switch(b.type) {
304  case Event::TypePolledLayer:
305  b.h = b.polledLayer->fd();
306  break;
307  case Event::TypeWinSock:
308  if((res = SOCKET_to_HANDLE(b.winsock, b.h)))
309  return res;
310  if((res = setEvents(b.winsock, b.h, events)))
311  WSACloseEvent(b.h);
312  return res;
313 # ifdef STORED_HAVE_ZMQ
314  case Event::TypeZmqSock: {
315  SOCKET s = INVALID_SOCKET; // NOLINT(hicpp-signed-bitwise)
316  if((res = socket_to_SOCKET(b.zmqsock, s)))
317  return res;
318  if((res = SOCKET_to_HANDLE(s, b.h)))
319  return res;
320  if((res = setEvents(s, b.h, events)))
321  WSACloseEvent(b.h);
322  return res;
323  }
324  case Event::TypeZmq: {
325  SOCKET s = b.zmq->fd();
326  if((res = SOCKET_to_HANDLE(s, b.h)))
327  return res;
328  if((res = setEvents(s, b.h, events)))
329  WSACloseEvent(b.h);
330  return res;
331  }
332 # endif
333  default:;
334  }
335 #endif
336 
337  return 0;
338 }

411 #ifdef STORED_POLL_WFMO
412 Poller::Result const& Poller::poll(long timeout_us, bool UNUSED_PAR(suspend)) {
413  errno = 0;
414 
415  m_lastEvents.clear();
416  if(m_events.empty()) {
417  errno = EFAULT;
418  return m_lastEvents;
419  }
420 
421  m_lastEventsH.clear();
422 
423  for(std::deque<Event>::iterator it = m_events.begin(); it != m_events.end(); ++it)
424  if(it->h)
425  m_lastEventsH.push_back(it->h);
426 
427  stored_assert(m_lastEventsH.size() < MAXIMUM_WAIT_OBJECTS);
428 
429  // First try.
430  DWORD res = WaitForMultipleObjectsEx((DWORD)m_lastEventsH.size(), &m_lastEventsH[0], FALSE, timeout_us >= 0 ? (DWORD)((timeout_us + 999L) / 1000L) : INFINITE, TRUE);
431 
432  HANDLE h = nullptr;
433  DWORD index = 0;
434  events_t revents = 0;
435 
436  while(true) {
437  if(res == WAIT_TIMEOUT) {
438  errno = EAGAIN;
439  break;
440  } else if(res == WAIT_IO_COMPLETION) {
441  // Completion routine is executed.
442  // Just retry.
443  errno = EINTR;
444  break;
445  } else if(res >= WAIT_OBJECT_0 && res < WAIT_OBJECT_0 + m_lastEventsH.size()) {
446  h = m_lastEventsH[index = res - WAIT_OBJECT_0];
447  revents = 0;
448  EVENTS_SET(revents, PollIn);
449  EVENTS_SET(revents, PollOut);
450  } else if(res >= WAIT_ABANDONED_0 && res < WAIT_ABANDONED_0 + m_lastEventsH.size()) {
451  h = m_lastEventsH[index = res - WAIT_ABANDONED_0];
452  revents = 0;
453  EVENTS_SET(revents, PollErr);
454  } else {
455  errno = EINVAL;
456  break;
457  }
458 
459  for(std::deque<Event>::iterator it = m_events.begin(); it != m_events.end(); ++it)
460  if(it->h == h) {
461 #ifdef STORED_HAVE_ZMQ
462  int zmq_events = ZMQ_POLLIN | ZMQ_POLLOUT; // NOLINT(hicpp-signed-bitwise)
463  size_t len = sizeof(zmq_events);
464  switch(it->type) {
465  case Event::TypeZmqSock:
466  if(zmq_getsockopt(it->zmqsock, ZMQ_EVENTS, &zmq_events, &len))
467  revents = 0;
468  revents &= (events_t)zmq_events;
469  WSAResetEvent(it->h);
470  break;
471  case Event::TypeZmq:
472  if(zmq_getsockopt(it->zmq->socket(), ZMQ_EVENTS, &zmq_events, &len))
473  revents = 0;
474  revents &= (events_t)zmq_events;
475  WSAResetEvent(it->h);
476  break;
477  default:;
478  }
479 #endif
480 
481  if(it->type == Event::TypeWinSock)
482  WSAResetEvent(it->h);
483 
484  if(revents) {
485  m_lastEvents.push_back(*it);
486  m_lastEvents.back().events = revents;
487  }
488  break;
489  }
490 
491  // WaitForMultipleObjects() only returns the first handle that was finished.
492  // But we would like to return them all. Check the rest too.
493  m_lastEventsH.erase(m_lastEventsH.begin() + (int)index);
494  if(m_lastEventsH.empty())
495  break;
496 
497  res = WaitForMultipleObjects((DWORD)m_lastEventsH.size(), &m_lastEventsH[0], FALSE, 0);
498  }
499 
500  if(m_lastEvents.empty() && !errno)
501  errno = EAGAIN;
502 
503  return m_lastEvents;
504 }
505 #endif

508 Poller::Result const& Poller::poll(long timeout_us, bool UNUSED_PAR(suspend)) {
509  m_lastEvents.resize(16); // Some arbitrary number. re-poll() to get the rest.
510 
511  int res = zmq_poller_wait_all(m_poller, &m_lastEvents[0], (int)m_lastEvents.size(), timeout_us >= 0 ? (timeout_us + 999L) / 1000L : -1L);
512 
513  if(res < 0) {
514  errno = zmq_errno();
515  m_lastEvents.clear();
516  } else if(res == 0) {
517  errno = EAGAIN;
518  m_lastEvents.clear();
519  } else {

528 #if defined(STORED_POLL_POLL) || defined(STORED_POLL_ZTH)
529 Poller::Result const& Poller::poll(long timeout_us, bool suspend) {
530  m_lastEvents.clear();
531  if(m_events.empty()) {
532  errno = EFAULT;
533  return m_lastEvents;
534  }
535 
536  m_lastEventsFd.resize(m_events.size());
537  size_t i = 0;
538  for(std::deque<Event>::iterator it = m_events.begin(); it != m_events.end(); ++it, ++i) {
539  pollfd& e = m_lastEventsFd[i];
540 
541  it->pollfd = &e;
542  e.events = (short)it->events;
543 
544  switch(it->type) {
545  case Event::TypeFd:
546  e.fd = it->fd;
547 #if defined(STORED_POLL_ZTH) && defined(ZTH_HAVE_ZMQ)
548  e.socket = nullptr;
549 #endif
550  break;
551  case Event::TypePolledLayer:
552  e.fd = it->polledLayer->fd();
553 #if defined(STORED_POLL_ZTH) && defined(ZTH_HAVE_ZMQ)
554  e.socket = nullptr;
555  break;
556  case Event::TypeZmqSock:
557  e.fd = -1;
558  e.socket = it->zmqsock;
559  break;
560  case Event::TypeZmq:
561  e.fd = -1;
562  e.socket = it->zmq->socket();
563 #endif
564  break;
565  default:
566  errno = EINVAL;
567  return m_lastEvents;
568  }
569  }
570 
571 retry:
572  int res = 0;
573 
574  if(suspend) {
575  // Just suspend; not fiber-aware.
576  res =
577 #if defined(STORED_POLL_ZTH) && defined(ZTH_HAVE_LIBZMQ)
578  ::zmq_poll
579 #else
580  ::poll
581 #endif
582  (&m_lastEventsFd[0], (nfds_t)m_lastEventsFd.size(), (int)(timeout_us > 0 ? (timeout_us + 999L) / 1000L : timeout_us));
583  } else {
584  res =
585 #ifdef STORED_POLL_ZTH
586  zth::io
587 #endif
588  ::poll(&m_lastEventsFd[0], (nfds_t)m_lastEventsFd.size(), (int)(timeout_us > 0 ? (timeout_us + 999L) / 1000L : timeout_us));
589  }
590 
591  switch(res) {
592  case -1:
593  switch(errno) {
594  case EINTR:
595  case EAGAIN:
596  goto retry;
597  default:
598  return m_lastEvents;
599  }
600  case 0:
601  errno = EAGAIN;
602  return m_lastEvents;
603  default:;
604  }
605 
606  for(std::deque<Event>::iterator it = m_events.begin(); it != m_events.end(); ++it) {
607  if(it->pollfd && it->pollfd->revents) {
608  m_lastEvents.push_back(*it);
609  m_lastEvents.back().events = (events_t)it->pollfd->revents;
610  }
611  }
612 
613  return m_lastEvents;
614 }
615 #endif

617 #if defined(STORED_POLL_LOOP) || defined(STORED_POLL_ZTH_LOOP)
618 bool Poller::poll_once() {
619  if(!m_lastEvents.empty())
620  return true;
621 
622  for(std::deque<Event>::iterator it = m_events.begin(); it != m_events.end(); ++it) {
623  events_t revents = 0;
624  if(stored::poll_once(*it, revents))
625  // Error
626  continue;
627  if(!(revents & it->events))
628  // Not the events we requested
