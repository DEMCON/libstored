255  void rollback(void* snapshot, size_type size) {
256  if(!snapshot || !size) {
257  reset();
258  return;
259  }
260 
261  char* snapshot_ = static_cast<char*>(snapshot);
262 
263  // Find correct buffer.
264  while(!(snapshot_ >= m_buffer && snapshot_ < &m_buffer[bufferSize()]))
265  {
266  m_size = 0; // Don't care, will be recovered later on.
267  bufferPop();
268  stored_assert(m_buffer);
269  }
270 
271  // Recover pointers within this buffer.
272  stored_assert(size <= m_total);
273  m_total = size;
274  size_t size_ = (size_t)(snapshot_ - m_buffer);
275  stored_assert(size_ < bufferSize());
276  m_size = (size_type)size_;

318  stored_assert(m_buffer || m_old.empty());
319 
320  if(m_buffer) {
321  free(chunk(m_buffer)); // NOLINT(cppcoreguidelines-owning-memory,cppcoreguidelines-no-malloc)
322  m_buffer = nullptr;
323  }
324 
325  if(!m_old.empty()) {
326  m_buffer = m_old.back();
327  m_old.pop_back();
328  }

418  void reserve(size_t more) {
419  size_t new_cap = m_size + more;
420 
421  if(likely(new_cap <= bufferSize()))
422  return;
423 
424  if(m_buffer && m_size == 0) {
425  // Grow (realloc) is fine, as nobody is using the current buffer (if any).
426  bufferGrow(new_cap);
427  } else {

429  new_cap = more + spare; // plus some extra reserve space

470  template <typename T>
471  __attribute__((malloc,returns_nonnull,warn_unused_result))
472  T* alloc(size_t count = 1, size_t align = sizeof(T)) {
473  size_t size = count * sizeof(T);
474  if(unlikely(size == 0)) {
475  if(unlikely(!m_buffer))
476  reserve(spare);
477  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
478  return reinterpret_cast<T*>(m_buffer);
479  }
480 
481  align = std::max<size_t>(1, std::min(sizeof(void*), align));
482  size_t padding = align - m_size % align;
483  if(padding == align)
484  padding = 0;
485 
486  if(unlikely(m_total + size + padding < m_total)) {
487  // Wrap around -> overflow.
488 #ifdef __cpp_exceptions
489  throw std::bad_alloc();
490 #else
491  abort();
492 #endif
493  }
494 
495  size_t bs = bufferSize();
496  if(likely(m_size + padding <= bs)) {
497  // The padding (which may be 0) still fits in the buffer.
498  m_size = (size_type)(m_size + padding);
499  // Now reserve the size, which still may add a new chunk.
500  if(unlikely(m_size + size > bs))
501  // Reserve all we probably need, if we are reserving anyway.
502  reserve(std::max(max() - this->size(), size));
503  } else {
504  // Not enough room for the padding, let alone the size.
505  // Just create a new buffer, which has always the correct alignment.
506  bufferPush(std::max(max() - this->size(), size + spare));
507  }
508 
509  char* p = m_buffer + m_size;
510  m_size = (size_type)(m_size + size);
511 
512  // Do count the padding, even it was not allocated, as it might be
513  // required anyway if the buffers are coalesced.
514  m_total = (size_type)(m_total + padding + size);
515 
516  if(unlikely(m_total > m_max))
517  m_max = m_total;
518 
519 #ifdef STORED_HAVE_VALGRIND
520  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-cstyle-cast,hicpp-no-assembler)
521  (void)VALGRIND_MAKE_MEM_UNDEFINED(p, size);
522 #else
523  if(Config::Debug)
524  memset(p, 0xef, size);
525 #endif
526  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
527  return reinterpret_cast<T*>(p);
528  }
