174  template <typename R> struct saturated_cast_helper
175  {
176  template <typename T> __attribute__((pure)) static R cast(T value)
177  {
178  // Lower bound check
179  if(std::numeric_limits<R>::is_integer) {
180  if(!std::numeric_limits<T>::is_signed) {
181  // No need to check
182  } else if(!std::numeric_limits<R>::is_signed) {
183  if(value <= 0)
184  return 0;
185  } else {
186  // Both are signed.
187  if(static_cast<typename signedness_helper<T>::signed_type>(value) <= static_cast<typename signedness_helper<R>::signed_type>(std::numeric_limits<R>::min()))
188  return std::numeric_limits<R>::min();
189  }
190  } else {
191 #pragma GCC diagnostic push
192 #pragma GCC diagnostic ignored "-Woverflow" // This error triggers when R is integer, but this code path is not trigger then.
193  if(value <= -std::numeric_limits<R>::max())
194  return -std::numeric_limits<R>::max();
195 #pragma GCC diagnostic pop
196  }
197 
198  // Upper bound check
199  if(value > 0)
200  if(static_cast<typename signedness_helper<T>::unsigned_type>(value) >= static_cast<typename signedness_helper<R>::unsigned_type>(std::numeric_limits<R>::max()))
201  return std::numeric_limits<R>::max();
202 
203  // Default conversion
204  return static_cast<R>(value);
205  }
