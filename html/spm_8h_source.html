129  void rollback(void* snapshot) {
130  if(unlikely(!snapshot)) {
131  reset();
132  } else if(likely(static_cast<char*>(snapshot) >= m_buffer
133  && static_cast<char*>(snapshot) < &m_buffer[m_size]))
134  {
135  m_size = (size_t)(static_cast<char*>(snapshot) - static_cast<char*>(m_buffer));

138  (void)VALGRIND_MAKE_MEM_NOACCESS(&m_buffer[m_size], m_capacity - m_size);
139 #endif
140  }

144  void reserve(size_t more) {
145  size_t new_cap = m_size + more;
146 
147  if(likely(new_cap <= m_capacity))
148  return;
149 
150  void* p;
151  if(m_size == 0) {
152  // clang-analyzer-unix.API: clang-tidy thinks new_cap can be 0, but that's not true.
153  // NOLINTNEXTLINE(cppcoreguidelines-no-malloc,cppcoreguidelines-owning-memory,clang-analyzer-unix.API)
154  p = realloc(m_buffer, new_cap);
155  } else {
156  // realloc() may move the memory, which makes all previously alloced pointers
157  // invalid. Push current buffer on m_old and alloc a new buffer.
158  m_old.push_back(m_buffer);
159  new_cap = more * 2 + sizeof(void*) * 8; // plus some extra reserve space
160  // NOLINTNEXTLINE(cppcoreguidelines-no-malloc,cppcoreguidelines-owning-memory)
161  p = malloc(new_cap);
162  m_size = 0;
163  }
164 
165  if(unlikely(!p))
166  throw std::bad_alloc();

168  m_buffer = static_cast<char*>(p);
